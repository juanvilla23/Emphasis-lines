# Crear matriz de pesos W como array de NumPy
W = np.zeros((len(A), len(A)), dtype=np.float64)
sigma = 1

for i in range(len(A)):
    for j in range(len(A)):
        if A[i,j] == 1: 
            diferencia = H_i_gorro[i] - H_i_gorro[j]
            # Fórmula correcta: exp(-((h_i - h_j)^2) / (2 * sigma^2))
            W[i,j] = np.exp(-(diferencia**2) / (2 * sigma**2))
        else:
            W[i,j] = 0

print(f"Número de aristas con peso no cero: {np.count_nonzero(W)}")

import numpy as np

def levenshtein_distance(s1, s2):
    rows = len(s1) + 1
    cols = len(s2) + 1
    
    dist_matrix = np.zeros((rows, cols), dtype=int)

    # Inicializar primera fila y columna
    for i in range(rows):
        dist_matrix[i, 0] = i
    for j in range(cols):
        dist_matrix[0, j] = j

    # Llenar matriz
    for i in range(1, rows):
        for j in range(1, cols):

            if s1[i-1] == s2[j-1]:
                cost = 0
            else:
                cost = 1

            dist_matrix[i, j] = min(
                dist_matrix[i-1, j] + 1,      # borrar
                dist_matrix[i, j-1] + 1,      # insertar
                dist_matrix[i-1, j-1] + cost  # sustituir
            )

    return dist_matrix[rows-1, cols-1]

Conclsuiones:

¿Por qué los resultados pueden diferir entre Jaccard, Levenshtein y Euclidiana?

Los resultados difieren, ya que cada uno de los algoritmos varía tanto en los atributos utilizados como en la métrica aplicada, ya que cada distancia usa una definición diferente de similitud: Jaccard compara la intersección de conjuntos, evaluando coincidencias de géneros, Levenshtein mide transformaciones de texto, hallando similitud narrativa, y la distancia Euclidiana evalúa separación geométrica entre vectores numéricos asociados a variables económicas. Por esto, a pesar de tener los mismos datos, las métricas producen diferentes vecinos más cercanos

¿si fuera a recomendar películas que distancia usaría? ¿por qué?

La métrica que utilizaría para recomendar películas similares es la distancia de Levenshtein, dado que nos genera una metrica sobre la columna overview, esta desición la tomamos teniendo en cuenta lo siguiente:

Las variables como presupuesto, revenue y popularidad reflejan el desempeño comercial de una película, no su contenido. Dos películas pueden tener presupuestos y popularidad similar y ser completamente diferentes. Por otra parte tenemos que la distancia de Jaccard mejora, ya que toma en cuenta los generos, los que si capturan información relevante sobre la película, pero sigue siendo demasiado general ya que dos películas pueden compartir géneros y aún asi tener tramas y públicos objetivos muy diferentes. Finalmente consideramos que entre las opciones vistas la que guarda más información sobre como es la película es el overview, que habla sobre trama personajes y a nuestro parecer es una medida más representativa del contenido real.
